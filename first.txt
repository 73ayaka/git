import streamlit as st

# Streamlitのタイトル
st.title('色でわかる性格診断')

# 説明文
st.write('好きな色を選んで、あなたの性格を診断しましょう！')

# 色の選択肢
colors = ['赤', '青', '緑', '黄色', '紫', '黒', '白']

# 好きな色を選んでもらう
favorite_color = st.selectbox('一番好きな色を選んでください', colors)

# 結果ボタン
if st.button('診断結果を見る'):
    st.write("診断結果")

    # 好きな色に基づく性格診断結果
    if favorite_color == '赤':
        st.write("あなたは情熱的でエネルギッシュな性格です！チャレンジ精神旺盛で、リーダーシップを発揮する場面が多いでしょう。")
    elif favorite_color == '青':
        st.write("あなたは冷静で落ち着いた性格です。思慮深く、周りの人から信頼される存在です。")
    elif favorite_color == '緑':
        st.write("あなたは穏やかで安定した性格です。調和を大切にし、人々との調和を保つのが得意です。")
    elif favorite_color == '黄色':
        st.write("あなたは明るく陽気な性格です。楽観的でクリエイティブな面があり、周囲に元気を与えます。")
    elif favorite_color == '紫':
        st.write("あなたはミステリアスで独特な性格です。感性が豊かで、芸術的な才能に恵まれています。")
    elif favorite_color == '黒':
        st.write("あなたは強い意志を持ち、内面的に自立した性格です。決断力があり、困難にも動じないタイプです。")
    elif favorite_color == '白':
        st.write("あなたは純粋で誠実な性格です。シンプルな生活を好み、正直さを大切にしています。")

# フッター
st.write("※ この診断はあくまで簡易的なものです。楽しんでください！")
